```java
@FXML
private void manejarLogin() {
    String usuario = usuarioField.getText().trim();
    String contraseña = contraseñaField.getText();
    String backend = backendCombo.getValue();
    
    // Validar entrada
    if (!Validaciones.esCampoRequerido(usuario)) {
        Alertas.mostrarError("Error", "Usuario requerido");
        return;
    }
    
    if (!Validaciones.esCampoRequerido(contraseña)) {
        Alertas.mostrarError("Error", "Contraseña requerida");
        return;
    }
    
    // Usar Task para no bloquear UI
    Task<ResultadoLogin> task = new Task<ResultadoLogin>() {
        @Override
        protected ResultadoLogin call() throws Exception {
            ServicioBancario servicio = ServicioFactory.crearServicio(backend);
            return servicio.autenticar(usuario, contraseña);
        }
    };
    
    task.setOnSucceeded(e -> {
        ResultadoLogin resultado = task.getValue();
        ContextoBackend.setUsuarioActual(resultado.getUsuario());
        ContextoBackend.setServicioActual(ServicioFactory.crearServicio(backend));
        navegarAMain();
    });
    
    task.setOnFailed(e -> {
        Throwable ex = task.getException();
        LOGGER.error("Error en login", ex);
        Alertas.mostrarError("Login Fallido", 
            "No se pudo conectar. Intente más tarde.");
    });
    
    new Thread(task).start();
}

private void navegarAMain() {
    try {
        FXMLLoader loader = new FXMLLoader(
            getClass().getResource("/views/MainView.fxml")
        );
        Parent root = loader.load();
        Scene scene = new Scene(root);
        Stage stage = (Stage) usuarioField.getScene().getWindow();
        stage.setScene(scene);
        stage.show();
    } catch (IOException e) {
        LOGGER.error("Error al cargar MainView", e);
    }
}
```

### Interfaz

```java
public interface ServicioBancario {
    ResultadoAutenticacion autenticar(String usuario, String contraseña)
        throws ExcepcionServicio;
    
    List<Movimiento> obtenerMovimientos(String cuenta)
```

### Implementación SOAP

```java
public class ServicioSoapJava implements ServicioBancario {
    
    private static final Logger LOGGER = 
        LoggerFactory.getLogger(ServicioSoapJava.class);
    
    private final String urlSoap;
    private final int timeoutConexion;
    
    public ServicioSoapJava(String urlSoap, int timeoutConexion) {
        this.urlSoap = urlSoap;
        this.timeoutConexion = timeoutConexion;
    }
    
    @Override
    public ResultadoAutenticacion autenticar(String usuario, String contraseña)
        throws ExcepcionServicio {
        
        try {
            LOGGER.debug("Autenticando usuario: {}", usuario);
            
            // Generar request SOAP
            String soapRequest = generarSoapLogin(usuario, contraseña);
            
            // Ejecutar request
            String soapResponse = ejecutarSoap(soapRequest);
            
            // Parsear response
            ResultadoAutenticacion resultado = 
                AdaptadorResponseSoap.parsearAutenticacion(soapResponse);
            
            LOGGER.info("Usuario autenticado exitosamente: {}", usuario);
            
            return resultado;
            
        } catch (ExcepcionSoapFault e) {
            LOGGER.error("SOAP Fault en autenticación", e);
            throw new ExcepcionServicio("Error del servidor", e);
        } catch (Exception e) {
            LOGGER.error("Error en autenticación", e);
            throw new ExcepcionServicio("No se pudo conectar", e);
        }
    }
    
    private String generarSoapLogin(String usuario, String contraseña) {
        return """
            <?xml version="1.0" encoding="UTF-8"?>
            <soap:Envelope xmlns:soap="http://schemas.xmlsoap.org/soap/envelope/">
                <soap:Body>
                    <loginRequest>
                        <usuario>${usuario}</usuario>
                        <contraseña>${contraseña}</contraseña>
                    </loginRequest>
                </soap:Body>
            </soap:Envelope>
            """.replace("${usuario}", usuario)
               .replace("${contraseña}", contraseña);
    }
    
    private String ejecutarSoap(String soapRequest) 
        throws IOException, ExcepcionSoapFault {
        
        URL url = new URL(urlSoap);
        HttpURLConnection conn = (HttpURLConnection) url.openConnection();
        
        conn.setRequestMethod("POST");
        conn.setRequestProperty("Content-Type", "text/xml");
        conn.setConnectTimeout(timeoutConexion);
        conn.setReadTimeout(timeoutConexion);
        
        try (OutputStream os = conn.getOutputStream()) {
            os.write(soapRequest.getBytes(StandardCharsets.UTF_8));
        }
        
        int responseCode = conn.getResponseCode();
        if (responseCode != HttpURLConnection.HTTP_OK) {
            throw new IOException("HTTP " + responseCode);
        }
        
        try (InputStream is = conn.getInputStream()) {
            return new String(is.readAllBytes(), StandardCharsets.UTF_8);
        }
    }
}
```

## Estructura de un Servicio REST

```java
public class ServicioRestJava implements ServicioBancario {
    
    private static final Logger LOGGER = 
        LoggerFactory.getLogger(ServicioRestJava.class);
    
    private final String urlRest;
    private final int timeoutConexion;
    private final Gson gson;
    
    public ServicioRestJava(String urlRest, int timeoutConexion) {
        this.urlRest = urlRest;
        this.timeoutConexion = timeoutConexion;
        this.gson = new Gson();
    }
    
    @Override
    public List<Movimiento> obtenerMovimientos(String cuenta)
        throws ExcepcionServicio {
        
        try {
            LOGGER.debug("Obteniendo movimientos de cuenta: {}", cuenta);
            
            URL url = new URL(urlRest + "/movimientos?cuenta=" + 
                URLEncoder.encode(cuenta, StandardCharsets.UTF_8));
            
            HttpURLConnection conn = (HttpURLConnection) url.openConnection();
            conn.setRequestMethod("GET");
            conn.setConnectTimeout(timeoutConexion);
            conn.setReadTimeout(timeoutConexion);
            
            int responseCode = conn.getResponseCode();
            
            if (responseCode == HttpURLConnection.HTTP_OK) {
                try (InputStream is = conn.getInputStream()) {
                    String response = new String(is.readAllBytes(), 
                        StandardCharsets.UTF_8);
                    List<Movimiento> movimientos = 
                        AdaptadorResponseRest.parsearMovimientos(response);
                    LOGGER.info("Movimientos obtenidos: {}", 
                        movimientos.size());
                    return movimientos;
                }
            } else if (responseCode == HttpURLConnection.HTTP_NOT_FOUND) {
                LOGGER.warn("Cuenta no encontrada: {}", cuenta);
                throw new ExcepcionCuentaNoEncontrada("Cuenta: " + cuenta);
            } else {
                throw new IOException("HTTP " + responseCode);
            }
            
        } catch (ExcepcionCuentaNoEncontrada e) {
            throw new ExcepcionServicio(e.getMessage(), e);
        } catch (Exception e) {
            LOGGER.error("Error obteniendo movimientos", e);
            throw new ExcepcionServicio("No se pudieron obtener datos", e);
        }
    }
}
```

## Patrón Factory

```java
public class ServicioFactory {
    
    private static final Logger LOGGER = 
        LoggerFactory.getLogger(ServicioFactory.class);
    
    private static final String PROPIEDAD_SOAP_JAVA = "soap.java.url";
    private static final String PROPIEDAD_REST_JAVA = "rest.java.url";
    private static final String PROPIEDAD_SOAP_DOTNET = "soap.dotnet.url";
    private static final String PROPIEDAD_REST_DOTNET = "rest.dotnet.url";
    private static final String PROPIEDAD_TIMEOUT = "timeout.connection";
    
    private static final Properties PROPIEDADES = cargarPropiedades();
    
    public enum TipoServicio {
        SOAP_JAVA("SOAP Java"),
        REST_JAVA("REST Java"),
        SOAP_DOTNET("SOAP .NET"),
        REST_DOTNET("REST .NET");
        
        private final String etiqueta;
        
        TipoServicio(String etiqueta) {
            this.etiqueta = etiqueta;
        }
        
        public String getEtiqueta() {
            return etiqueta;
        }
        
        public static TipoServicio desdEtiqueta(String etiqueta) {
            for (TipoServicio tipo : values()) {
                if (tipo.etiqueta.equals(etiqueta)) {
                    return tipo;
                }
            }
            throw new IllegalArgumentException("Tipo desconocido: " + etiqueta);
        }
    }
    
    public static ServicioBancario crearServicio(String tipo) 
        throws ExcepcionServicio {
        
        TipoServicio tipoServicio = TipoServicio.desdEtiqueta(tipo);
        int timeout = obtenerTimeout();
        
        return switch (tipoServicio) {
            case SOAP_JAVA -> new ServicioSoapJava(
                PROPIEDADES.getProperty(PROPIEDAD_SOAP_JAVA), timeout);
                
            case REST_JAVA -> new ServicioRestJava(
                PROPIEDADES.getProperty(PROPIEDAD_REST_JAVA), timeout);
                
            case SOAP_DOTNET -> new ServicioSoapDotNet(
                PROPIEDADES.getProperty(PROPIEDAD_SOAP_DOTNET), timeout);
                
            case REST_DOTNET -> new ServicioRestDotNet(
                PROPIEDADES.getProperty(PROPIEDAD_REST_DOTNET), timeout);
        };
    }
    
    private static Properties cargarPropiedades() {
        Properties props = new Properties();
        try (InputStream is = ServicioFactory.class
            .getResourceAsStream("/application.properties")) {
            if (is != null) {
                props.load(is);
            }
        } catch (IOException e) {
            LOGGER.error("Error cargando propiedades", e);
        }
        return props;
    }
    
    private static int obtenerTimeout() {
        String timeout = PROPIEDADES.getProperty(PROPIEDAD_TIMEOUT, "5000");
        return Integer.parseInt(timeout);
    }
}
```

## Validaciones Centralizadas

```java
public class Validaciones {
    
    private static final Logger LOGGER = 
        LoggerFactory.getLogger(Validaciones.class);
    
    public static boolean esCuentaValida(String cuenta) {
        if (cuenta == null || cuenta.trim().isEmpty()) {
            return false;
        }
        // Solo números
        return cuenta.matches("\\d{8}");
    }
    
    public static boolean esImporteValido(BigDecimal monto) {
        return monto != null && monto.compareTo(BigDecimal.ZERO) > 0;
    }
    
    public static boolean esCampoRequerido(String campo) {
        return campo != null && !campo.trim().isEmpty();
    }
    
    public static ResultadoValidacion validarLogin(String usuario, 
        String contraseña) {
        
        if (!esCampoRequerido(usuario)) {
            return new ResultadoValidacion(false, "Usuario requerido");
        }
        
        if (!esCampoRequerido(contraseña)) {
            return new ResultadoValidacion(false, "Contraseña requerida");
        }
        
        if (usuario.length() < 3) {
            return new ResultadoValidacion(false, 
                "Usuario debe tener al menos 3 caracteres");
        }
        
        return new ResultadoValidacion(true, "");
    }
}
```

```java
public class Alertas {

        LoggerFactory.getLogger(Alertas.class);
```

## Contexto Global (Sesión)

```java
public class ContextoBackend {
    
    private static final Logger LOGGER = 
        LoggerFactory.getLogger(ContextoBackend.class);
    
    private static Usuario usuarioActual;
    private static ServicioBancario servicioActual;
    private static String backendActual;
    
    public static void inicializarSesion(Usuario usuario, 
        ServicioBancario servicio, String backend) {
        
        usuarioActual = usuario;
        servicioActual = servicio;
        backendActual = backend;
        
        LOGGER.info("Sesión iniciada para usuario: {} con backend: {}", 
            usuario.getNombre(), backend);
    }
    
    public static void cerrarSesion() {
        usuarioActual = null;
        servicioActual = null;
        backendActual = null;
        LOGGER.info("Sesión cerrada");
    }
    
    public static Usuario getUsuarioActual() {
        return usuarioActual;
    }
    
    public static ServicioBancario getServicioActual() {
        return servicioActual;
    }
    
    public static String getBackendActual() {
        return backendActual;
    }
    
    public static boolean sesionActiva() {
        return usuarioActual != null && servicioActual != null;
    }
}
```
